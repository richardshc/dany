			if(id > 0) {
				$(`button#answers-${id}`).click(() => pasteContent(id, threadData, true));
			} еще {
				$(`button#answers-${id}`).click(() => pasteContent(id, threadData, false));
			}
		});
	});
});

функция addButton(имя, идентификатор) {

}

функция buttonsMarkup(кнопки) {
return `<div class="select_answer">${кнопки
  .карта(
	(btn, i) =>
	  `<button id="answers-${i}" class="button--primary button ` +
	  `rippleButton" style="margin:5px"><span class="button-text">${btn.title}</button>`,
  )
  .join(")}</div>`;
}

функция pasteContent(id, data = {}, send = false) {
	const template = Handlebars.compile(buttons[id].content);
	if ($('.fr-element.fr-view p').text() === ") $('.fr-element.fr-view p').empty();

	$('span.fr-placeholder').empty();

	$('a.overlay-titleCloser').trigger('click');

	if(send == true){
		editThreadData(buttons[id].prefix, buttons[id].status);

	}
}

функция editThreadData(префикс, pin = false) {

	const threadTitle = $('.p-title-value')[0].lastChild.textContent;
